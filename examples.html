1. Login Form with Validation

async function handleLogin(email, password) {
    // Validation
    if (!email || !email.includes('@')) {
        AndroidNotification.error('Invalid Email', 'Please enter a valid email address');
        return;
    }
    if (password.length < 6) {
        AndroidNotification.warning('Weak Password', 'Password must be at least 6 characters');
        return;
    }

    try {
        const response = await fetch('/api/login', {
            method: 'POST',
            body: JSON.stringify({ email, password })
        });
        
        if (response.ok) {
            AndroidNotification.success('Welcome! ✓', `Logged in as ${email}`);
        } else {
            AndroidNotification.error('Login Failed ✕', 'Invalid credentials');
        }
    } catch (error) {
        AndroidNotification.error('Network Error ✕', 'Could not connect to server');
    }
}

5. File Upload with Progress

async function uploadFileWithProgress(file) {
    const formData = new FormData();
    formData.append('file', file);
    
    try {
        const xhr = new XMLHttpRequest();
        
        xhr.upload.addEventListener('progress', (e) => {
            const progress = Math.round((e.loaded / e.total) * 100);
            console.log(`Upload progress: ${progress}%`);
        });
        
        xhr.addEventListener('load', () => {
            if (xhr.status === 200) {
                AndroidNotification.success(
                    'Upload Complete ✓',
                    `${file.name} uploaded successfully`
                );
            }
        });
        
        xhr.addEventListener('error', () => {
            AndroidNotification.error('Upload Failed ✕', 'Could not upload file');
        });
        
        xhr.open('POST', '/api/upload');
        xhr.send(formData);
    } catch (error) {
        AndroidNotification.error('Error ✕', 'Upload failed: ' + error.message);
    }
}

9. Error Recovery Pattern

async function retryWithNotification(fn, maxRetries = 3) {
    for (let attempt = 1; attempt <= maxRetries; attempt++) {
        try {
            return await fn();
        } catch (error) {
            if (attempt === maxRetries) {
                AndroidNotification.error(
                    'Operation Failed ✕',
                    `Failed after ${maxRetries} attempts`
                );
                throw error;
            } else {
                AndroidNotification.warning(
                    'Retrying... ⟳',
                    `Attempt ${attempt}/${maxRetries}`
                );
                await new Promise(resolve => setTimeout(resolve, 1000 * attempt));
            }
        }
    }
}

3. Error Handling with Recovery Options

function handleError(error, retryFn) {
    const errorMessages = {
        'NETWORK_ERROR': 'No internet connection. Check your network.',
        'TIMEOUT': 'Request took too long. Please try again.',
        'SERVER_ERROR': 'Server error. Please try again later.',
        'NOT_FOUND': 'Resource not found.',
    };

    const message = errorMessages[error.code] || 'An error occurred';
    
    if (retryFn) {
        Notification.error('Error!', message + ' Retrying...', 3000);
        setTimeout(() => retryFn(), 3000);
    } else {
        Notification.error('Error!', message);
    }
}

4. Batch Operations with Progress

async function processBatch(items) {
    let completed = 0;
    let failed = 0;

    for (const item of items) {
        try {
            await processItem(item);
            completed++;
            Notification.info('Progress', 
                `${completed}/${items.length} items processed`, 2000);
        } catch (error) {
            failed++;
            console.error(`Failed to process ${item.id}`);
        }
    }

    if (failed === 0) {
        Notification.success('Done!', 
            `All ${completed} items processed successfully`);
    } else {
        Notification.warning('Partial Success', 
            `${completed} succeeded, ${failed} failed`);
    }
}

5. Form Auto-Save with Debounce

let saveTimeout;

function handleFormChange(formData) {
    // Clear previous timeout
    clearTimeout(saveTimeout);
    
    // Show saving status
    Notification.info('Saving...', 'Your changes are being saved', 0);
    
    // Debounce the save request
    saveTimeout = setTimeout(() => {
        fetch('/api/save', {
            method: 'POST',
            body: JSON.stringify(formData)
        })
        .then(() => {
            Notification.clearAll();
            Notification.success('Saved', 'Your changes have been saved', 2000);
        })
        .catch(() => {
            Notification.clearAll();
            Notification.error('Save Failed', 'Could not save changes');
        });
    }, 1000); // Save after 1 second of no changes
}

6. Multi-step Process with Validation

const steps = [
    { name: 'Validate', fn: validateData },
    { name: 'Upload', fn: uploadFile },
    { name: 'Process', fn: processFile },
    { name: 'Complete', fn: finishProcess }
];

async function executeSteps() {
    for (let i = 0; i < steps.length; i++) {
        const step = steps[i];
        Notification.info(`Step ${i + 1}/${steps.length}`, 
            `${step.name}...`, 0);
        
        try {
            await step.fn();
        } catch (error) {
            Notification.clearAll();
            Notification.error('Process Failed', 
                `Failed at step: ${step.name}`);
            return;
        }
    }
    
    Notification.clearAll();
    Notification.success('Complete', 'All steps finished successfully!');
}
