💻 How to Use Native Notifications

// Request permission first
await Notification.requestPermission();

// Send native Android notification
Notification.sendNative('Title', {
    body: 'Notification message',
    icon: '/icon-192x192.png',
    badge: '/badge-72x72.png',
    tag: 'notification-id'
});

// Or use the helper
AndroidNotification.success('Done!', 'Task completed');
AndroidNotification.error('Error', 'Something failed');
AndroidNotification.warning('Warning', 'Check this');
AndroidNotification.info('Info', 'FYI message');
